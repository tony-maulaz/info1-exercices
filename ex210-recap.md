## Q5
Créer une boucle qui affiche les multiple de 3 entre 25 et 42 ?

```c
#include <stdio.h>

int main()
{
    int val1 = 10;
    int val2 = 7;
    
    int res1 = val1 % val2;
    printf("Res1 : %d\n", res1);
    
    printf("\nBoucle\n");
    for(int i=0; i<30; i++){
        if( i%7 == 0 )
            printf("%d\n",i);
    }
    
    printf("\nRes2 : %d\n", val2 > 0 ? val2 : -val2);

    return 0;
}
```
